    тіеТђЂУДётѕњ(Dynamic Programming№╝їDP)¶

    С╗ЦжЌ«жбў "уѕгтЈ░жўХ" СИ║СЙІ, жбўуЏ«тдѓСИІ:

    You are climbing a staircase. It takes n steps to reach the top.

    Each time you can either climb 1 or 2 steps. In how many distinct ways can you climb to the top?

    Example 1:

    Input: n = 2
Output: 2
Explanation: There are two ways to climb to the top.
1. 1 step + 1 step
2. 2 steps

    Example 2:

    Input: n = 3
Output: 3
Explanation: There are three ways to climb to the top.
1. 1 step + 1 step + 1 step
2. 1 step + 2 steps
3. 2 steps + 1 step

    Constraints:

    • 1 <= n <= 45

    тіЏТЅБ: https://leetcode.cn/problems/climbing-stairs/.

    Т│еТёЈ: СИІТќЄСИГуџё "ТГЦ", СИјСИіжЮбжбўуЏ«жЄїуџё "step" тљФС╣ЅСИЇтљї.

    ТхІУ»ЋућеСЙІ¶

    fn main() {
    assert_eq!(climb_stairs(2), 2);
    assert_eq!(climb_stairs(3), 3);
    assert_eq!(climb_stairs(5), 8);
    assert_eq!(climb_stairs(18), 4181);
}

    Тџ┤тіЏТљюу┤б (Brute-force search)¶

    жбўуЏ«УдЂТ▒ѓт«ъуј░тЄйТЋ░ climb_stairs, С╗ЦСИІСй┐ућетЄйТЋ░ \(f\) УАеуц║.

    • С╗југг 0 у║ДтЈ░жўХтѕ░УЙЙугг 1 у║ДтЈ░жўХ, тЁ▒СИђуДЇуѕгТ│Ћ, тЇ│СИђТгАуѕгСИђу║ДтЈ░жўХ. тЏаТГц \(f(1) = 1\).
    • С╗југг 0 у║ДтЈ░жўХтѕ░УЙЙугг 2 у║ДтЈ░жўХ, тЁ▒СИцуДЇуѕгТ│Ћ, СИђТгАуѕгСИђу║ДтЈ░жўХТѕќСИцу║ДтЈ░жўХ. тЏаТГц \(f(2) = 1 + 1 = 2\).

    \(f(1)\) тњї \(f(2)\) уџёу╗ЊТъютЙѕтЦйУ«Ау«Ќ, тЏаСИ║жЃйтЈфжюђУдЂуѕгСИђТгАтЈ░жўХ. СйєС╗ј \(f(3)\) т╝ђтДІ, УЄ│т░ЉжюђУдЂуѕгСИцТгАтЈ░жўХ.

    • С╗југг 1 у║ДтЈ░жўХтѕ░УЙЙугг 3 у║ДтЈ░жўХ, тЁ▒СИђуДЇуѕгТ│Ћ, тЇ│СИђТгАуѕгСИцу║ДтЈ░жўХ.
    • С╗југг 2 у║ДтЈ░жўХтѕ░УЙЙугг 3 у║ДтЈ░жўХ, тЁ▒СИђуДЇуѕгТ│Ћ, тЇ│СИђТгАуѕгСИђу║ДтЈ░жўХ.

    тЏаТГцУ«Ау«Ќ \(f(3)\) тЈ»С╗ЦтѕЕућетѕ░ \(f(1)\) тњї \(f(2)\) уџётђ╝. тЙЌтѕ░уіХТђЂУйгуД╗Тќ╣уеІ:

    \[ f(n) = f(n-1) + f(n-2) \]
    pub fn climb_stairs(n: i32) -> i32 {
    assert!(1 <= n && n <= 45);
    if n == 1 || n == 2 {
        n
    } else {
        climb_stairs(n - 1) + climb_stairs(n - 2)
    }
}

    ТЌХжЌ┤тцЇТЮѓт║дСИ║ \(O(2^n)\), уЕ║жЌ┤тцЇТЮѓт║дСИ║ \(O(n)\).

    У»Цу«ЌТ│Ћт«ъуј░уџёТюђтцДжЌ«жбўтюеС║јтЁХТЌХжЌ┤тцЇТЮѓт║дТў»ТїЄТЋ░у║ДтѕФуџё.

    У«░т┐єтїќТљюу┤б(Memoization)¶

    СИіжЮбТџ┤тіЏТљюу┤буџёт«ъуј░тГўтюетцДжЄЈжЄЇтцЇТђДУ«Ау«Ќ:

    • \(f(n) = f(n-1) + f(n-2)\)
    • \(f(n-1) = f(n-1-1) + f(n-2-1) = f(n-2) + f(n-3)\)

    тЁХСИГ \(f(n-2)\) т░єУбФжЄЇтцЇУ«Ау«Ќ, тЈ»С╗ЦжђџУ┐Єу╝ЊтГўтЄйТЋ░уџёУ«Ау«Ќу╗ЊТъюТЮЦжЂ┐тЁЇжЄЇтцЇУ«Ау«Ќ. 

    pub fn climb_stairs(n: i32) -> i32 {
    assert!(1 <= n && n <= 45);
    climb_stairs_inner(n, &mut HashMap::from([(1, 1), (2, 2)]))
}

fn climb_stairs_inner(n: i32, cache: &mut HashMap<i32, i32>) -> i32 {
    // ТБђТЪЦу╝ЊтГўСИГТў»тљдти▓тГўтюеУ«Ау«Ќу╗ЊТъю, УІЦТюЅуЏ┤ТјЦУ┐ћтЏъу╗ЊТъю, жЂ┐тЁЇжЄЇтцЇУ«Ау«Ќ
    if let Some(value) = cache.get(&n) {
        // у╝ЊтГўтЉйСИГ
        return *value;
    }
    let result = climb_stairs_inner(n - 1, cache) + climb_stairs_inner(n - 2, cache);
    // у╝ЊтГўУ«Ау«Ќу╗ЊТъю
    cache.insert(n, result);
    result
}

    ТЌХжЌ┤тцЇТЮѓт║дСИ║ \(O(n)\), уЕ║жЌ┤тцЇТЮѓт║дСИ║ \(O(n)\).

    У┐ЎТаиУ«Ау«Ќ \(f(n)\) тЈфжюђУдЂУ«Ау«Ќ \(f(n - 1), f(n - 2) ..., f(1)\). ТЅђС╗ЦТЌХжЌ┤тцЇТЮѓт║дСИ║ \(O(n)\).

    жђњТје¶

    • \(f(n) = f(n-1) + f(n-2)\)
    • \(f(n-1) = f(n-2) + f(n-3)\)
    • \(f(n-2) = f(n-3) + f(n-4)\)
    • \(...\)
    • \(f(2) = 2\)
    • \(f(1) = 1\)
    \[ \begin{align} У«Ау«Ќ f(n) \rightarrow У«Ау«Ќ f(n-1), f(n-2) \rightarrow У«Ау«Ќ f(2), f(1) \end{align} \]
    \[ \begin{align} У«Ау«Ќ f(2), f(1) \rightarrow У«Ау«Ќ f(n-1), f(n-2) \rightarrow У«Ау«Ќ f(n) \end{align} \]

    УІЦтѕЕућежђњтйњу«ЌТ│ЋУЄфСИіУђїСИІуџёТќ╣Т│Ћ(Top-down approach)СЙЮТгАТ▒ѓтђ╝(1)С╝џС║ДућЪжЄЇтцЇУ«Ау«Ќ, СйєТћ╣тЈўТ▒ѓтђ╝жА║т║Ј, Сй┐ућеУЄфСИІУђїСИіуџёТќ╣Т│Ћ(Bottom-up approach)Т▒ѓтђ╝(2)тѕЎтЈ»С╗ЦжЂ┐тЁЇУ»ЦжЌ«жбў.

    pub fn climb_stairs(n: i32) -> i32 {
    assert!(1 <= n && n <= 45);
    let mut f = vec![0, 1];
    for _ in 0..n as usize {
        f.push(f[f.len() - 1] + f[f.len() - 2]);
    }
    f[n as usize + 1]
}

    ТЌХжЌ┤тцЇТЮѓт║дСИ║ \(O(n)\), уЕ║жЌ┤тцЇТЮѓт║дСИ║ \(O(n)\).

    Тћ╣У┐Џтљјуџёу«ЌТ│ЋСИЇтєЇжюђУдЂжђџУ┐Єу╝ЊтГўТЮЦжЂ┐тЁЇжЄЇтцЇУ«Ау«Ќ, УђїСИћСИЇтєЇСй┐ућежђњтйњ.

    жђџУ┐ЄУДѓт»ЪтЁѕтЅЇуџётЁгт╝Ј, тЈ»С╗ЦтЈЉуј░угг n УАїуџётђ╝уГЅС║југг n-1 УАїуџётђ╝тіаСИіугг n-2 УАїуџётђ╝.
    тЏаТГцтЈфжюђУдЂтГўтѓетЅЇСИцТгАуџёУ«Ау«Ќу╗ЊТъю:

    pub fn climb_stairs(n: i32) -> i32 {
    assert!(1 <= n && n <= 45);
    let mut f0 = 0;
    let mut f1 = 1;
    for _ in 0..n {
        let f2 = f1 + f0;
        f0 = f1;
        f1 = f2;
    }
